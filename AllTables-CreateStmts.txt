All tables - Create Statements for Athena and/or Hive/SparkSQL:

CREATE EXTERNAL TABLE IF NOT EXISTS workshopdb.analytics1 (
  `playerip` string,
  `handle` string,
  `email` string,
  `uuid` string,
  `playerid` bigint,
  `country` string,
  `useragent` string,
  `datestamp` string,
  `walletbalance` double,
  `playerlevel` string,
  `status` string 
)
ROW FORMAT SERDE 'org.openx.data.jsonserde.JsonSerDe'
WITH SERDEPROPERTIES (
  'serialization.format' = '1'
) LOCATION 's3://{WorkshopBucket}/analytics1/'
TBLPROPERTIES ('has_encrypted_data'='false');
